<h2>
컴포넌트 명명규칙 > 대문자로 시작하는 파스칼 표기법 (꼭)
</h2>

```jsx
# App.jsx (컴포넌트)
--------------------------------------------------------------
import Child from './Child' // 컴포넌트 import 
function Hello() {
  return <h1>Hello</h1>
}

function App() {

  return {
    <>
      <Hello/> /* 내부 컴포넌트 호출 */
      <h2>World></h2>
      <Child/> /* 외부 컴포넌트 호출 */
    </>
  }
} 
export default App
--------------------------------------------------------------

# Child.jsx (하위 컴포넌트)
--------------------------------------------------------------
function Child() {
  return (
    <p> child </p>
  )
}
export default Child
--------------------------------------------------------------
```

<h2>
  3가지 형태의 컴포넌트
</h2>

```jsx
# Componenets.jsx
--------------------------------------------------------------
import React, { Component } from 'react'

class ClassComp extends Component { /* 오늘날에는 잘 사용되고있진 않음 */
  render () {
    return <p>Class Comp</p>;
  }
} 

/* 아래 두가지 함수형 컴포넌트가 일반적으로 사용됨 */
function FunctionComp() {
  return <p>Function Comp</p>;
}
/* 최근엔 간결하고 this 바인딩으로부터 자유로운 화살표 함수가 널리 쓰인다고 한다 */
const ArrowFunctionComp = () => {
  return <p>Arrow Function Comp</p>;
}

export { ClassComp, FunctionComp, ArrowFunctionComp };
--------------------------------------------------------------

# App.jsx (3가지 형태 컴포넌트 호출)
--------------------------------------------------------------
import {ClassComp, FunctionComp, ArrowFunctionComp} from './Components'

function Hello() {
  return <h1>Hello</h1>
}

function App() {

  return {
    <>
      <Hello/> /* 내부 컴포넌트 호출 */
      <h2>World></h2>
      <ClassComp/>
      <FunctionComp/>
      <ArrowFunctionComp/>
    </>
  }
} 
export default App
--------------------------------------------------------------
```
